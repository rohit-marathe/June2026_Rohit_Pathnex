# Ansible - Install + Start Multiple Services

---
- name: Install and enabled services 
  hosts: all
  become: yes

  tasks:
    - name: Install services
      yum: 
        name: 
            - httpd
            - firewalld
        state: present 

    - name: Enabled and start services
      services:
        name: "{{ item }}"
        state: started
        enabled: yes
      loop:
        - httpd 
        - firewalld

# Jenkins Pipeline - Deploy to Dynamic Environment 

pipeline {
    agent any
    environment {
        INSTITUTE_NAME = "Pathnex"
        TEAM = "Devops"
    }
    parameters {
        string(name: 'ENVIRONMENT', defaultValue: 'dev', description: 'Deployment Environment')
    }
    stages {
        stage('Deploy') {
            steps {
                sh 'echo "Deploying $INSTITUTE_NAME app to $ENVIRONMENT for $TEAM team "'
            }
        }
    }
}

# Ansible - Implementing SSH Key-Based Authentication 

- name: Implement SSH Key Authentication
  hosts: all
  become: yes
  tasks:
    - name: Generate SSH Key Pair 
      command: ssh-keygen -t rsa -b 2048 -f /home/user/.ssh/id_rsa -N ""
    - name: Add SSH Public Key to authorized_keys
      authorized_keys:
        user: user
        state: present 
        key: "{{ lookup{'file', '/home/user/.ssh/id_rsa.pub'} }}"

# Kubernetes - Network Policices for Security

apiVersion: networking.k8s.io/v1
kind: NetworkPolicy
metadata:
    name: pathnex-web-policy 
spec:
    podSelector:
        matchLabels:
            app: pathnex-web-app 
    policyTypes:
        - Ingress
        - Egress
    Ingress:
        - form: 
            - PodSelector:
                matchLabels:
                    app: pathnex-api 
    Egress:
        - to 
            - podSelector:
                matchLabels:
                    app: pathnex-database

# Jenkinsfile - Secure the Pipeline With Secrets


pipeline {
    agent any
    environment {
        AWS_ACCESS_KEY_ID = credentials('aws-access-key')
        AWS_SECRET_ACCESS_KEY = credentials('aws-secret-key')
    }
    stages {
        stage('Build') {
            steps {
                echo 'Building Docker Image...'
                docker.build('pathnex-web-app')
            }
        }
        stage('Deploy') {
            steps {
                echo 'Deploying to AWS EC2...'
                sh 'aws ec2 run-instances --image-id ami-0abcd1234 --count 1 --instance-type t2.micro'
            }
        }
    }
}

# Gitlab CI/CD - Use Gitalb Secrets For Secure Deployment

    stage: deploy
