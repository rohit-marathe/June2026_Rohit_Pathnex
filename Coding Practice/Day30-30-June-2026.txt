# Terraform - main.tf 

provider "aws" {
    region = "ap-south-1"
}

resource "aws_vpc" "pathnex" {
    cidr_block = "10.0.0.0/16"

    tags = {
        Name = "Pathnex-VPC"
    }
}

resource "aws_subnet" "public" {
    vpc_id = aws_vpc.pathnex.id 
    cidr_block = 10.0.0.0/24
    availability_zone = "ap-south-1a"
    map_public_ip_on_launch = true 
}

resource "aws_security_group" "web" {
    name = "pathnex-sg"
    vpc_id = aws_vpc.pathnex.id 

    ingress {
        from_port = 22
        to_port = 22
        protocol = "tcp"
        cidr_block = ["0.0.0.0/0"]
    }

    ingress {
        from_port = 80
        to_port = 80
        protocol = "tcp"
        cidr_block = ["0.0.0.0/0"]
    }

    egress {
        from_port = 0
        to_port = 0
        protocol = "-1"
        cidr_block = ["0.0.0.0/0"]
    }
}

resource "aws_instance" "app" {
    ami = "ami-xxxxxxxx"
    instance_type = "t2.micro"
    subnet_id = aws_subnet.public.id 
    vpc_security_group_ids = [aws_security_group.web.id]

    tags = {
        Name = "Pathnex-Server"
        Project = "Pathnex"
    }
}

# Jenkins Pipeline 

pipeline {
    agent any 

    environment {
        PROJECT = "Pathnex"
        ENV = "Prod"
    }
    stages {
        
        stage('Terraform') {
            steps {
                sh '''
                terraform init 
                terraform apply -auto-apporve
            }
        }
        stage('Build') {
            steps {
                sh 'docker build -t pathnex:latest .'
            }
        }
        stage('Push Image') {
            steps {
                sh 'docker push <dockerhub-user>/pathnex:latest'
            }
        }
        stage('Deploy Kubernetes') {
            steps {
                sh '''
                kubectl apply -f deployment.yml
                helm upgrade --install pathnex ./pathnex 
            }
        }
        stage('Monitoring') {
            steps {
                sh '''
                kubectl get pods
                kubectl get svc 
                kubectl top pods 
                '''
            }
        }
    }
}
