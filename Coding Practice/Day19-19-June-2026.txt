# Ansible - Deploy app using git

---
- name: Deploy app from Git for Pathnex
  hosts: all
  become: yes

  tasks:
    - name: Install git
      yum:
        name: git
        state: present 
    
    - name: Clone repository
      git:
        repo: "https://github.com/pathnex/app.git"
        dest: "/opt/pathnex-app"

# Kubernetes - Deployment Rolling Update (Detailed)

apiVersion: apps/v1
kind: Deployment 
metadata:
    name: pathnex-rollout 
spec:
    replicas: 3
    strategy:
        type: RollingUpdate
        rollingUpdate:
            maxSurge: 2
            maxUnavailable: 1
    
    selector:
        matchLabels:
            app: rollout
    
    template:
        matchLabels:
            app: rollout
    template:
        metadata:
            labels:
                app: rollout 
    spec:
        containers:
            - name: app
              image: nginx

    agent any

        INSTITUTE_NAME = "Pathnex"

# Ansible - Setup Prometheus For Monitoring

- name: Install Prometheus For Monitoring
  hosts: all
  become: yes
  tasks:
    - name: Install Prometheus dependencies 
      yum:
        name:
         - Prometheus
         - prometheus-node-exporter
    
    - name: Start Prometheus Service 
      services:
        name: prometheus
        state: started
        enabled: yes

# Kubernetes - Deploy Fluentd for Logging 

apiVersion: apps/v1
kind: Deployment
metadata:
    name: fluentd 

spec:
    replicas: 1
    selector:
        matchLabels:
            app: fluentd
    template:
        metadata:
            labels:
                app: fluentd
    
    spec:
        containers:
            - name: fluentd
              image: fluentd/fluentd
              ports:
                - containerPort: 24224

---

apiVersion: v1
kind: service
metadata:
    name: fluentd
spec:
    selector:
        app: fluentd
    ports:
        - protocol: TCP
          ports: 24224

#Jenkinsfile - Integrating Prometheus Monitoring in Pipeline

pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                echo 'Building Docker image...'
                docker.build('pathnex-web-app')
            }
        }
        stage('Test') {
            steps {
                echo 'Running Tests....'
            }
        }
        stage('Monitor Deployment') {
            steps {
                script {
                    sh 'kubectl apply -f kubernetes/deployment.yaml'
                    sh 'kubectl get pods --watch'
                }
            }
        }
    }
}
