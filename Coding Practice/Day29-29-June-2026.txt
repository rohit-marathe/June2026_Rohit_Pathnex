# Kubernetes - PersistentVolume (PV)

    name: nginx-pv

    capacity:
        storage: 1Gi
    accessModes:
        - ReadWriteOnce

# Jenkins Pipeline 

pipeline {
    agent any 

    environmet {
        INSTITUTE_NAME = "Pathnex"
        TEAM = "DevOps"
        ENV = "prod"
        PROJECT = "Pathnex-Training"
    }

    stages {
        stage('Infrastructure') {
            steps {
                sh '''
                    echo "Applying Terraform..."
                    terraform init 
                    terraform apply -auto-approve
            }
        }
        stage('App Deployment') {
            steps {
                sh '''
                    echo "Running Ansible Playbooks..."
                    ansible-playbook -i inventory deploy.yml 

                    echo "Applying Kubernetes manifests..."
                    kuebctl apply -f k8s/
                ''' 
            }
        }
        stage('Monitoring checks') {
            steps {
                sh '''
                    echo "CPU Usage:"
                    top -bn1 | grep "Cpu(s)"

                    echo "Memory Usage:"
                    free -h 

                    echo "Disk Usage:"
                    hostname -I 

                    echo "Top Process:"
                    ps -eo pid,comm,%cpu --sort=-%cpu | head -n 6
                '''
            }
        }
    }

    post {
        always {
            echo "Pipeline executed for ${PROJECT} by ${TEAM} at ${INSTITUTE_NAME} in ${ENV} environmet"
        }
    }
}
